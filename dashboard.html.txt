    } else {
      data.forEach(item => {
        const tr = document.createElement('tr');
        tr.innerHTML = `

          <td>${item.aduan}</td>
          <td>${item.tanggal}</td>
          <td><span class="status-${item.status.toLowerCase()}">${item.status}</span></td>
